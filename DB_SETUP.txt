DROP TABLE IF EXISTS LOGIN;
CREATE TABLE LOGIN (username TEXT PRIMARY KEY, userID);

DROP TABLE IF EXISTS AUTH;
CREATE TABLE AUTH (userID TEXT PRIMARY KEY, secretKey TEXT);

DROP TABLE IF EXISTS USER;
CREATE TABLE USER (userID TEXT PRIMARY KEY, username TEXT, secretKey TEXT);

DROP TABLE IF EXISTS WORKOUT; -- SHOULD ADD CREATEDBY TO WORKOUT
CREATE TABLE WORKOUT (workoutID TEXT PRIMARY KEY, userID TEXT, state TEXT, modified INT, timeStarted REAL, timeCompleted REAL, totalTime REAL);

DROP TABLE IF EXISTS EXERCISE;
CREATE TABLE EXERCISE (exerciseID TEXT PRIMARY KEY, workoutID TEXT, createdByID TEXT, exerciseName TEXT, exerciseDescription TEXT, state TEXT, exerciseCatagory TEXT, exerciseMuscleGroup TEXT, timeStarted REAL, timeCompleted REAL, totalTime REAL)
